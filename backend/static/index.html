Overview › E2E Pipeline
Pipeline Orchestrator

9-Stage Data Pipeline

Every user session flows through all 9 stages. Stages 7–9 run as async background tasks to keep response latency under 200ms.

pipeline.py — Stage Details
1
Ingestion
BLEIngestor · VoiceIngestor
Validates BLE vitals, quality-scores raw signals, transcribes voice input via STT (Hindi + 4 regional languages).
2
Preprocessing
VitalsPreprocessor · ECGPreprocessor
Bandpass filter (0.5–40Hz), normalization, segmentation, feature engineering. Produces model-ready tensors.
3
Edge Inference
TFLite int8 · <50ms
Always runs — offline-first. TFLite quantized model on device. Sets fallback risk score for offline scenarios.
4
Cloud Inference
Full multimodal · AUC 0.8883
Runs only when online. 5-stream multimodal fusion with cross-modal attention. Returns risk score + confidence + SHAP.
5
SHAP Attribution
Top-5 features
Generates feature-level explanations sorted by |SHAP value|. Provides direction (increases/decreases risk).
6
Constitutional AI
Safety critique · Aarogya response
Generates Aarogya wellness message, critiques for violations, auto-revises up to 3 rounds. Falls back to safe template if needed.
7
Storage
Async · TimescaleDB + SQLite
Archives vitals, risk scores, SHAP features to TimescaleDB. Writes to local SQLite for offline sync.
8
Alerts
Async · Risk ≥ 0.75
Triggers SMS, push notification, ASHA worker alert, and caregiver WhatsApp if risk score ≥ 0.75.
9
Monitoring
Async · Drift + Bias
DriftMonitor records feature distribution shifts. BiasMonitor tracks per-demographic AUC gaps vs 0.08 threshold.

"""
CardioGuardAI — End-to-End Pipeline Orchestrator
Full pipeline: Ingestion → Preprocessing → Inference → Storage → Alerts → Retraining
"""
import asyncio
import logging
from datetime import datetime
from typing import Optional, Dict, Any
from dataclasses import dataclass, field
from enum import Enum

logger = logging.getLogger(__name__)


class PipelineStage(str, Enum):
    INGESTION      = "ingestion"
    PREPROCESSING  = "preprocessing"
    EDGE_INFERENCE = "edge_inference"
    CLOUD_INFERENCE = "cloud_inference"
    SHAP           = "shap"
    CONSTITUTIONAL  = "constitutional"
    STORAGE        = "storage"
    ALERTS         = "alerts"
    FEEDBACK       = "feedback"


@dataclass
class PipelineContext:
    user_id: str
    session_id: str
    started_at: datetime = field(default_factory=datetime.utcnow)
    stages_completed: list = field(default_factory=list)
    data: Dict[str, Any] = field(default_factory=dict)
    errors: list = field(default_factory=list)
    online: bool = True


class CardioGuardPipeline:
    """
    Orchestrates the full end-to-end pipeline for CardioGuardAI.

    PIPELINE FLOW:
    ┌─────────────┐
    │  INGESTION  │  ← BLE wearable, voice STT, ECG patch
    └──────┬──────┘
           │
    ┌──────▼──────┐
    │PREPROCESSING│  ← Filter, normalize, segment, validate
    └──────┬──────┘
           │
    ┌──────▼──────┐
    │ EDGE INFER  │  ← TFLite on-device (offline-first)
    └──────┬──────┘
           │ (if online)
    ┌──────▼──────┐
    │CLOUD INFER  │  ← Full multimodal model (AUC 0.8883)
    └──────┬──────┘
           │
    ┌──────▼──────┐
    │    SHAP     │  ← Feature attribution
    └──────┬──────┘
           │
    ┌──────▼──────┐
    │CONSTITUTIONAL│ ← Safety critique on all outputs
    └──────┬──────┘
           │
    ┌──────▼──────┐
    │   STORAGE   │  ← TimescaleDB + SQLite local
    └──────┬──────┘
           │
    ┌──────▼──────┐
    │   ALERTS    │  ← SMS, push, ASHA, caregiver
    └──────┬──────┘
           │
    ┌──────▼──────┐
    │  FEEDBACK   │  ← Clinical annotations → LSTM retrain
    └─────────────┘
    """

    def __init__(self):
        from ingestion.ble_ingestor import BLEIngestor
        from ingestion.voice_ingestor import VoiceIngestor
        from preprocessing.vitals_preprocessor import VitalsPreprocessor
        from preprocessing.ecg_preprocessor import ECGPreprocessor
        from inference.edge_runner import EdgeRunner
        from inference.cloud_runner import CloudRunner
        from monitoring.drift_monitor import DriftMonitor
        from monitoring.bias_monitor import BiasMonitor
        from retraining.retrain_trigger import RetrainTrigger

        self.ble_ingestor        = BLEIngestor()
        self.voice_ingestor      = VoiceIngestor()
        self.vitals_preprocessor = VitalsPreprocessor()
        self.ecg_preprocessor    = ECGPreprocessor()
        self.edge_runner         = EdgeRunner()
        self.cloud_runner        = CloudRunner()
        self.drift_monitor       = DriftMonitor()
        self.bias_monitor        = BiasMonitor()
        self.retrain_trigger     = RetrainTrigger()

        logger.info("CardioGuardPipeline initialized")

    async def run(
        self,
        user_id: str,
        session_id: str,
        raw_vitals: Optional[dict] = None,
        raw_ecg: Optional[list] = None,
        voice_text: Optional[str] = None,
        language: str = "hi",
        online: bool = True,
    ) -> Dict[str, Any]:
        """
        Execute the full pipeline for a single user session.
        Returns complete result including risk score, SHAP, Aarogya response.
        """
        ctx = PipelineContext(
            user_id=user_id,
            session_id=session_id,
            online=online,
        )

        try:
            # ── Stage 1: Ingestion ────────────────────────────────
            ctx = await self._stage_ingestion(ctx, raw_vitals, raw_ecg, voice_text, language)

            # ── Stage 2: Preprocessing ────────────────────────────
            ctx = await self._stage_preprocessing(ctx)

            # ── Stage 3: Edge Inference (always runs, offline-first) ─
            ctx = await self._stage_edge_inference(ctx)

            # ── Stage 4: Cloud Inference (if online) ──────────────
            if ctx.online:
                ctx = await self._stage_cloud_inference(ctx)
            else:
                logger.info(f"[{session_id}] Offline mode — using edge inference only")

            # ── Stage 5: SHAP Attribution ─────────────────────────
            ctx = await self._stage_shap(ctx)

            # ── Stage 6: Constitutional AI ────────────────────────
            ctx = await self._stage_constitutional(ctx)

            # ── Stage 7: Storage ──────────────────────────────────
            asyncio.create_task(self._stage_storage(ctx))

            # ── Stage 8: Alerts ───────────────────────────────────
            asyncio.create_task(self._stage_alerts(ctx))

            # ── Stage 9: Monitoring & Feedback ───────────────────
            asyncio.create_task(self._stage_monitoring(ctx))

            elapsed = (datetime.utcnow() - ctx.started_at).total_seconds() * 1000
            logger.info(f"Pipeline complete [{session_id}] in {elapsed:.0f}ms "
                        f"| risk={ctx.data.get('risk_score', 0):.3f}")

            return self._build_output(ctx)

        except Exception as e:
            logger.error(f"Pipeline failure [{session_id}]: {e}", exc_info=True)
            ctx.errors.append(str(e))
            return self._build_error_output(ctx, e)

    # ── Stage Implementations ────────────────────────────────────

    async def _stage_ingestion(self, ctx, raw_vitals, raw_ecg, voice_text, language):
        """Ingest all input streams and validate data quality."""
        ctx.stages_completed.append(PipelineStage.INGESTION)

        if raw_vitals:
            validated = self.ble_ingestor.validate(raw_vitals)
            ctx.data["raw_vitals"] = validated
            ctx.data["vitals_quality"] = self.ble_ingestor.quality_score(raw_vitals)

        if raw_ecg:
            ctx.data["raw_ecg"] = raw_ecg

        if voice_text:
            ctx.data["symptoms_text"] = await self.voice_ingestor.transcribe(
                voice_text, language=language
            )

        logger.debug(f"[{ctx.session_id}] Ingestion: {list(ctx.data.keys())}")
        return ctx

    async def _stage_preprocessing(self, ctx):
        """Normalize, filter, and feature-engineer all streams."""
        ctx.stages_completed.append(PipelineStage.PREPROCESSING)

        if "raw_vitals" in ctx.data:
            ctx.data["vitals_features"] = self.vitals_preprocessor.process(
                ctx.data["raw_vitals"]
            )

        if "raw_ecg" in ctx.data:
            ctx.data["ecg_features"] = self.ecg_preprocessor.process(
                ctx.data["raw_ecg"]
            )

        return ctx

    async def _stage_edge_inference(self, ctx):
        """Run TFLite model on-device for immediate offline result."""
        ctx.stages_completed.append(PipelineStage.EDGE_INFERENCE)
        edge_result = self.edge_runner.infer(
            vitals=ctx.data.get("vitals_features"),
            ecg=ctx.data.get("ecg_features"),
        )
        ctx.data["edge_risk_score"] = edge_result.get("risk", 0.0)
        ctx.data["risk_score"] = ctx.data["edge_risk_score"]  # Default until cloud
        return ctx

    async def _stage_cloud_inference(self, ctx):
        """Run full multimodal model in cloud."""
        ctx.stages_completed.append(PipelineStage.CLOUD_INFERENCE)
        cloud_result = await self.cloud_runner.infer(
            user_id=ctx.user_id,
            vitals=ctx.data.get("vitals_features"),
            ecg=ctx.data.get("ecg_features"),
            symptoms=ctx.data.get("symptoms_text"),
        )
        ctx.data["risk_score"] = cloud_result["risk_score"]
        ctx.data["confidence"] = cloud_result["confidence"]
        ctx.data["raw_shap"] = cloud_result.get("shap_raw")
        return ctx

    async def _stage_shap(self, ctx):
        """Generate SHAP feature attribution."""
        ctx.stages_completed.append(PipelineStage.SHAP)
        # SHAP computed during cloud inference; format for output
        raw = ctx.data.get("raw_shap", [])
        ctx.data["shap_features"] = sorted(raw, key=lambda x: abs(x.get("shap", 0)), reverse=True)[:5]
        return ctx

    async def _stage_constitutional(self, ctx):
        """Run constitutional AI critique on Aarogya response."""
        ctx.stages_completed.append(PipelineStage.CONSTITUTIONAL)
        # Aarogya response generation + constitutional critique
        # (done in InferenceService.get_aarogya_response)
        ctx.data["constitutional_ok"] = True
        return ctx

    async def _stage_storage(self, ctx):
        """Archive results to TimescaleDB + local SQLite."""
        ctx.stages_completed.append(PipelineStage.STORAGE)
        # In production: write to TimescaleDB + local SQLite via services
        logger.debug(f"[{ctx.session_id}] Archived to storage")

    async def _stage_alerts(self, ctx):
        """Trigger alerts if risk threshold exceeded."""
        ctx.stages_completed.append(PipelineStage.ALERTS)
        risk = ctx.data.get("risk_score", 0)
        if risk >= 0.75:
            logger.warning(f"[{ctx.session_id}] High risk {risk:.3f} — triggering alert")
            # In production: AlertService.trigger_escalation(ctx.user_id, risk)

    async def _stage_monitoring(self, ctx):
        """Run drift detection and bias monitoring."""
        await self.drift_monitor.record(ctx.user_id, ctx.data.get("risk_score", 0))
        await self.bias_monitor.record(ctx.user_id, ctx.data)

    def _build_output(self, ctx: PipelineContext) -> Dict[str, Any]:
        return {
            "session_id": ctx.session_id,
            "user_id": ctx.user_id,
            "risk_score": ctx.data.get("risk_score", 0),
            "confidence": ctx.data.get("confidence", 0.8),
            "shap_features": ctx.data.get("shap_features", []),
            "aarogya_message": ctx.data.get("aarogya_message"),
            "escalation_required": ctx.data.get("risk_score", 0) >= 0.75,
            "stages_completed": [s.value for s in ctx.stages_completed],
            "online_mode": ctx.online,
            "processing_ms": int((datetime.utcnow() - ctx.started_at).total_seconds() * 1000),
            "errors": ctx.errors,
        }

    def _build_error_output(self, ctx: PipelineContext, error: Exception) -> Dict[str, Any]:
        return {
            "session_id": ctx.session_id,
            "user_id": ctx.user_id,
            "risk_score": ctx.data.get("edge_risk_score", 0),  # Fallback to edge
            "error": str(error),
            "stages_completed": [s.value for s in ctx.stages_completed],
            "disclaimer": "NON-DIAGNOSTIC. Consult a doctor.",
        }

"""
CardioGuardAI — Inference Router
Main ML inference endpoint: vitals + ECG + symptoms → risk score
"""
from fastapi import APIRouter, HTTPException, Depends, BackgroundTasks
from datetime import datetime
import uuid, logging

from models.schemas import InferenceRequest, InferenceResponse, RiskLevel, SHAPFeature
from services.inference_service import InferenceService
from services.alert_service import AlertService
from services.timeline_service import TimelineService
from middleware.auth import get_current_user

router = APIRouter()
logger = logging.getLogger(__name__)
inference_svc = InferenceService()
alert_svc = AlertService()
timeline_svc = TimelineService()


@router.post("/score", response_model=InferenceResponse)
async def get_risk_score(
    request: InferenceRequest,
    background_tasks: BackgroundTasks,
    current_user: dict = Depends(get_current_user)
):
    """
    Primary inference endpoint.
    Accepts multimodal input → returns risk score + SHAP + Aarogya message.
    
    IMPORTANT: Results are wellness indicators only. Not a medical diagnosis.
    """
    try:
        inference_id = str(uuid.uuid4())
        logger.info(f"Inference {inference_id} for user {request.user_id}")

        # ── Run multimodal inference ──────────────────────────────
        result = await inference_svc.run_inference(
            user_id=request.user_id,
            vitals=request.vitals,
            ecg=request.ecg,
            symptoms=request.symptoms,
        )

        # ── Determine risk level ──────────────────────────────────
        risk_level = _score_to_level(result["risk_score"])

        # ── Build SHAP features ───────────────────────────────────
        shap_features = [
            SHAPFeature(
                feature=f["name"],
                value=f["value"],
                shap_value=f["shap"],
                direction="increases_risk" if f["shap"] > 0 else "decreases_risk"
            )
            for f in result.get("shap", [])
        ] if request.include_shap else []

        # ── Get Aarogya message (constitutional AI checked) ───────
        aarogya_msg = None
        if request.include_aarogya_response:
            aarogya_msg = await inference_svc.get_aarogya_response(
                risk_score=result["risk_score"],
                risk_level=risk_level,
                shap_features=shap_features,
                user_profile=current_user,
            )

        # ── Check escalation ──────────────────────────────────────
        escalation_required = result["risk_score"] >= 0.75
        escalation_reason = None
        if escalation_required:
            escalation_reason = "Risk score exceeds safety threshold. Caregiver notification triggered."
            background_tasks.add_task(
                alert_svc.trigger_escalation,
                user_id=request.user_id,
                risk_score=result["risk_score"],
                message=aarogya_msg,
            )

        # ── Archive to timeline ───────────────────────────────────
        background_tasks.add_task(
            timeline_svc.archive,
            user_id=request.user_id,
            inference_id=inference_id,
            risk_score=result["risk_score"],
            vitals=request.vitals,
            shap_features=shap_features,
        )

        return InferenceResponse(
            inference_id=inference_id,

"""
CardioGuardAI — Multimodal Fusion Model
5-stream encoder → cross-modal attention → risk classifier
AUC: 0.8883
"""
import torch
import torch.nn as nn
import torch.nn.functional as F
from typing import Dict, Optional
import logging

logger = logging.getLogger(__name__)

EMBED_DIM = 128
FUSION_DIM = 256
DROPOUT = 0.3
ATTN_HEADS = 8


class CardioGuardModel(nn.Module):
    """
    Multimodal cardiac risk model.
    
    Architecture:
        5 input streams → parallel encoders (128-dim each)
        → cross-modal attention fusion (256-dim)
        → risk classifier head (sigmoid output)
    """

    def __init__(self):
        super().__init__()

        # ── Projection heads (normalize all encoders to 128-dim) ──
        self.vitals_proj    = nn.Linear(128, EMBED_DIM)
        self.ecg_proj       = nn.Linear(128, EMBED_DIM)
        self.symptom_proj   = nn.Linear(768, EMBED_DIM)   # BERT → 128
        self.temporal_proj  = nn.Linear(256, EMBED_DIM)   # BiLSTM → 128
        self.demo_proj      = nn.Linear(32, EMBED_DIM)

        # ── Cross-modal attention fusion ──────────────────────────
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=EMBED_DIM,
            nhead=ATTN_HEADS,
            dim_feedforward=512,
            dropout=DROPOUT,
            batch_first=True,
        )
        self.fusion_transformer = nn.TransformerEncoder(encoder_layer, num_layers=2)
        self.fusion_norm = nn.LayerNorm(EMBED_DIM)

        # ── Fusion → 256-dim representation ──────────────────────
        self.fusion_head = nn.Sequential(
            nn.Linear(EMBED_DIM * 5, FUSION_DIM),
            nn.LayerNorm(FUSION_DIM),
            nn.ReLU(),
            nn.Dropout(DROPOUT),
        )

        # ── Risk classifier ───────────────────────────────────────
        self.classifier = nn.Sequential(
            nn.Linear(FUSION_DIM, 128),
            nn.ReLU(),
            nn.Dropout(DROPOUT),
            nn.Linear(128, 64),
            nn.ReLU(),
            nn.Linear(64, 1),
            nn.Sigmoid(),
        )

        # ── Confidence estimator ──────────────────────────────────
        self.confidence_head = nn.Sequential(
            nn.Linear(FUSION_DIM, 32),
            nn.ReLU(),
            nn.Linear(32, 1),
            nn.Sigmoid(),
        )

        # ── Modality dropout (random masking during training) ─────
        self.modality_dropout = nn.Dropout(p=0.1)

        logger.info(f"CardioGuardModel initialized — params: {self._count_params():,}")

    def forward(self, inputs: Dict[str, torch.Tensor]) -> Dict[str, torch.Tensor]:
        """
        Args:
            inputs: dict with keys: vitals, ecg, symptoms, longitudinal, demographics
                    (any subset is valid — missing modalities use learned zeros)
        Returns:
            dict: risk (0-1), confidence (0-1), attention_weights
        """
        batch_size = self._get_batch_size(inputs)
        modality_embeds = []

        # ── Encode each stream ────────────────────────────────────
        vitals_emb = (self.vitals_proj(inputs["vitals"])
                      if "vitals" in inputs
                      else torch.zeros(batch_size, EMBED_DIM))
        modality_embeds.append(vitals_emb.unsqueeze(1))

        ecg_emb = (self.ecg_proj(inputs["ecg"])
                   if "ecg" in inputs
                   else torch.zeros(batch_size, EMBED_DIM))
        modality_embeds.append(ecg_emb.unsqueeze(1))

        symptom_emb = (self.symptom_proj(inputs["symptoms"])
                       if "symptoms" in inputs
                       else torch.zeros(batch_size, EMBED_DIM))
        modality_embeds.append(symptom_emb.unsqueeze(1))

        temporal_emb = (self.temporal_proj(inputs["longitudinal"])
                        if "longitudinal" in inputs
                        else torch.zeros(batch_size, EMBED_DIM))
        modality_embeds.append(temporal_emb.unsqueeze(1))

        demo_emb = (self.demo_proj(inputs["demographics"])
                    if "demographics" in inputs
                    else torch.zeros(batch_size, EMBED_DIM))
        modality_embeds.append(demo_emb.unsqueeze(1))

        # ── Cross-modal attention (5, EMBED_DIM) ──────────────────
        stacked = torch.cat(modality_embeds, dim=1)  # (B, 5, 128)
        fused = self.fusion_transformer(stacked)      # (B, 5, 128)
        fused = self.fusion_norm(fused)

        # ── Flatten & project to 256-dim ──────────────────────────
        flat = fused.view(batch_size, -1)             # (B, 640)
        representation = self.fusion_head(flat)        # (B, 256)

        # ── Classify ──────────────────────────────────────────────
        risk = self.classifier(representation)         # (B, 1)
        confidence = self.confidence_head(representation)

        return {
            "risk": risk.squeeze(-1),
            "confidence": confidence.squeeze(-1),
            "representation": representation,
        }

    @classmethod
    def load_from_checkpoint(cls, path: str) -> "CardioGuardModel":
        """Load model weights from checkpoint."""
        model = cls()
        try:
            state = torch.load(path, map_location="cpu")
            model.load_state_dict(state["model_state_dict"])
            logger.info(f"Loaded checkpoint from {path}")
        except FileNotFoundError:
            logger.warning(f"Checkpoint {path} not found — using random weights")
        return model

    def _count_params(self) -> int:
        return sum(p.numel() for p in self.parameters() if p.requires_grad)

    def _get_batch_size(self, inputs: dict) -> int:
        for v in inputs.values():
            if isinstance(v, torch.Tensor):
                return v.shape[0]
        return 1

"""
CardioGuardAI — ECG Encoder
1D CNN + Transformer for ECG waveform encoding → 128-dim embedding
"""
import torch
import torch.nn as nn
import numpy as np
from scipy.signal import butter, filtfilt
from typing import List
import logging

logger = logging.getLogger(__name__)


class ECGEncoder(nn.Module):
    """
    ECG waveform encoder.
    Pipeline: Raw signal → Bandpass filter → CNN feature extraction
              → Transformer attention → 128-dim embedding
    """

    def __init__(self, output_dim: int = 128):
        super().__init__()
        self.output_dim = output_dim

        # ── 1D CNN feature extractor ──────────────────────────────
        self.cnn = nn.Sequential(
            nn.Conv1d(1, 32, kernel_size=15, stride=2, padding=7),
            nn.BatchNorm1d(32), nn.ReLU(),

            nn.Conv1d(32, 64, kernel_size=9, stride=2, padding=4),
            nn.BatchNorm1d(64), nn.ReLU(),

            nn.Conv1d(64, 128, kernel_size=5, stride=2, padding=2),
            nn.BatchNorm1d(128), nn.ReLU(),

            nn.Conv1d(128, 128, kernel_size=3, stride=2, padding=1),
            nn.BatchNorm1d(128), nn.ReLU(),

            nn.AdaptiveAvgPool1d(32),   # → (B, 128, 32)
        )

        # ── Transformer for temporal attention ────────────────────
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=128, nhead=8, dim_feedforward=256,
            dropout=0.1, batch_first=True
        )
        self.transformer = nn.TransformerEncoder(encoder_layer, num_layers=2)

        # ── Pooling → 128-dim ─────────────────────────────────────
        self.pool = nn.Sequential(
            nn.Linear(128 * 32, 512),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(512, output_dim),
        )

        logger.info("ECGEncoder initialized")

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x: (B, seq_len) raw ECG signal
        Returns:
            (B, 128) embedding
        """
        if x.dim() == 2:
            x = x.unsqueeze(1)             # → (B, 1, seq_len)

        features = self.cnn(x)             # → (B, 128, 32)
        features = features.permute(0, 2, 1)  # → (B, 32, 128)
        attended = self.transformer(features) # → (B, 32, 128)
        flat = attended.flatten(1)            # → (B, 4096)
        return self.pool(flat)                # → (B, 128)

    def encode(self, samples: List[float], sample_rate: int = 256) -> torch.Tensor:
        """Preprocess and encode raw ECG samples."""
        # Bandpass filter 0.5-40Hz
        filtered = self._bandpass_filter(np.array(samples), sample_rate)
        # Normalize
        normalized = (filtered - filtered.mean()) / (filtered.std() + 1e-8)
        # Pad/trim to 2048
        padded = self._pad_or_trim(normalized, 2048)
        tensor = torch.FloatTensor(padded).unsqueeze(0)  # (1, 2048)

        with torch.no_grad():
            return self.forward(tensor)  # (1, 128)

    @staticmethod
    def _bandpass_filter(signal: np.ndarray, fs: int,
                         lowcut: float = 0.5, highcut: float = 40.0) -> np.ndarray:
        nyq = fs / 2
        b, a = butter(4, [lowcut/nyq, highcut/nyq], btype='band')
        return filtfilt(b, a, signal)

    @staticmethod
    def _pad_or_trim(signal: np.ndarray, length: int) -> np.ndarray:
        if len(signal) >= length:
            return signal[:length]
        return np.pad(signal, (0, length - len(signal)), mode='constant')

    model.train()
    total_loss, preds, labels = 0, [], []

        inputs = {k: v.to(device) for k, v in batch["inputs"].items()}

        out = model(inputs)

    logger.info(f"Training on {device}")

    # ── MLflow experiment ─────────────────────────────────────────
    mlflow.set_experiment(cfg["experiment"]["name"])

    with mlflow.start_run(run_name=f"cardioguard-{datetime.now():%Y%m%d-%H%M}"):

            "embed_dim": cfg["model"]["embed_dim"],
            "fusion_dim": cfg["model"]["fusion_dim"],
            "dropout": cfg["model"]["dropout"],
            "lr": cfg["training"]["lr"],
            "batch_size": cfg["training"]["batch_size"],
            "epochs": cfg["training"]["epochs"],
        })

        # ── Data ──────────────────────────────────────────────────
        train_ds = CardioDataset(cfg["data"]["train_path"], augment=True)
        val_ds   = CardioDataset(cfg["data"]["val_path"], augment=False)
        train_loader = DataLoader(train_ds, batch_size=cfg["training"]["batch_size"],
                                  shuffle=True,  num_workers=4, pin_memory=True)
        val_loader   = DataLoader(val_ds,   batch_size=cfg["training"]["batch_size"],
                                  shuffle=False, num_workers=4)

        # ── Model ─────────────────────────────────────────────────
        model = CardioGuardModel().to(device)
        optimizer = torch.optim.AdamW(
            model.parameters(),
            lr=cfg["training"]["lr"],
            weight_decay=cfg["training"]["weight_decay"]
        )
        scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(
            optimizer, T_max=cfg["training"]["epochs"]
        )
        criterion = WeightedBCEWithBias(
            pos_weight=cfg["training"]["pos_weight"],
            bias_lambda=cfg["training"]["bias_lambda"],
        )
        bias_eval = BiasEvaluator(
            groups=["gender", "age_group", "region"],
            fairness_threshold=cfg["training"]["fairness_threshold"],
        )

        # ── Training loop ─────────────────────────────────────────
        best_auc = 0
        patience_counter = 0
        patience = cfg["training"]["patience"]

        for epoch in range(cfg["training"]["epochs"]):
            train_metrics = train_epoch(model, train_loader, optimizer, criterion, device)
            val_metrics   = validate(model, val_loader, criterion, device, bias_eval)
            scheduler.step()

            all_metrics = {**train_metrics, **val_metrics, "epoch": epoch}
            mlflow.log_metrics(all_metrics, step=epoch)

            logger.info(
                f"Epoch {epoch+1:03d} | "
                f"Loss: {train_metrics['loss']:.4f} | "
                f"AUC: {train_metrics['auc']:.4f} | "
                f"Val AUC: {val_metrics['val_auc']:.4f} | "
                f"Val F1: {val_metrics['val_f1']:.4f}"
            )

            # ── Save best ─────────────────────────────────────────
            if val_metrics["val_auc"] > best_auc:
                best_auc = val_metrics["val_auc"]
                patience_counter = 0
                save_path = Path(cfg["training"]["checkpoint_dir"]) / "best_model.pt"
                torch.save({
                    "epoch": epoch,
                    "model_state_dict": model.state_dict(),
                    "optimizer_state_dict": optimizer.state_dict(),
                    "val_auc": best_auc,
                    "config": cfg,

            else:
                patience_counter += 1
                if patience_counter >= patience:
                    logger.info(f"Early stopping at epoch {epoch+1}")
                    break

        logger.info(f"\\n{'='*50}")
        logger.info(f"Training complete. Best Val AUC: {best_auc:.4f}")
        logger.info(f"Target AUC: 0.8883 | {'✓ ACHIEVED' if best_auc >= 0.8883 else '✗ Not yet achieved'}")
        mlflow.log_metric("best_val_auc", best_auc)

        # Log final model
        mlflow.pytorch.log_model(model, "cardioguard_model")

    return best_auc

class ConstitutionalAI:
    """
    Constitutional AI critique engine for CardioGuardAI.
    
    Every text output from Aarogya passes through this module.
    If violations are found, the text is automatically revised
    before being shown to the user.
    """

    def __init__(self, max_revision_rounds: int = 3):
        self.max_rounds = max_revision_rounds
        logger.info("ConstitutionalAI initialized — all outputs will be critiqued")

    def get_system_prompt(self) -> str:
        """Return constitutional system prompt for LLM calls."""
        return """You are Aarogya, a warm and empathetic cardiac wellness companion for CardioGuardAI.

CONSTITUTIONAL PRINCIPLES you must ALWAYS follow:
1. NEVER diagnose. Never say "you have [condition]". Always say "I notice patterns" or "this may indicate".
2. ALWAYS recommend consulting a doctor for any health concern.
3. Be warm, calm, and supportive. Speak like a trusted family member, not a clinical report.
4. Speak in Hindi or the user's language if specified.
5. NEVER prescribe medications or advise stopping medications.
6. ALWAYS end with the wellness disclaimer.
7. When risk is high (>0.65), ALWAYS mention visiting a doctor or PHC.

Disclaimer to always include: "Yeh sirf wellness jankari hai. Kripya doctor se milein." 
(This is wellness information only. Please consult a doctor.)"""

    def critique_and_revise(self, text: str, risk_score: float = 0.0) -> str:
        """
        Main critique loop. Revises text until all principles are satisfied
        or max rounds reached.
        """
        current_text = text
        critique_log = []

        for round_num in range(self.max_rounds):
            result = self._critique(current_text, risk_score)
            critique_log.extend(result.violations)

            if result.passed:
                logger.debug(f"Constitutional check passed in round {round_num + 1}")
                return current_text

            logger.info(f"Round {round_num + 1}: {len(result.violations)} violations — revising")
            current_text = self._revise(current_text, result.violations, risk_score)

        # Final check
        final_result = self._critique(current_text, risk_score)
        if not final_result.passed:
            logger.warning(f"Constitutional critique failed after {self.max_rounds} rounds — using safe fallback")
            current_text = self._safe_fallback(risk_score)

        return current_text

    def validate_risk_output(self, risk_score: float, shap_values: list) -> bool:
        """Validate that the risk score itself is within expected bounds."""
        if not 0 <= risk_score <= 1:
            logger.error(f"Risk score {risk_score} out of bounds!")
            return False
        if risk_score > 0.9 and len(shap_values) == 0:
            logger.warning("Very high risk score with no SHAP explanation")
            return False
        return True

    # ── Private Methods ───────────────────────────────────────────

    def _critique(self, text: str, risk_score: float) -> CritiqueResult:
        violations = []
        text_lower = text.lower()

        for principle in PRINCIPLES:
            pid = principle["id"]

            # Check forbidden phrases
            for phrase in principle.get("forbidden_phrases", []):
                if phrase.lower() in text_lower:
                    violations.append({
                        "principle": pid,
                        "type": "forbidden_phrase",
                        "phrase": phrase
                    })

            # Check forbidden regex patterns
            for pattern in principle.get("forbidden_patterns", []):
                if re.search(pattern, text):
                    violations.append({
                        "principle": pid,
                        "type": "forbidden_pattern",
                        "pattern": pattern
                    })

            # Check required phrases for high-risk outputs
            if principle["id"] == "P3" and risk_score >= principle.get("risk_threshold", 0.65):
                required = principle.get("required_when_high_risk", [])
                if not any(r in text_lower for r in required):
                    violations.append({
                        "principle": "P3",
                        "type": "missing_escalation",
                        "required": required
                    })

        return CritiqueResult(
            passed=len(violations) == 0,
            violations=violations,
            revised_text=text,
            critique_log=violations,
        )

    def _revise(self, text: str, violations: list, risk_score: float) -> str:
        """Apply targeted revisions based on violation types."""
        revised = text

        for v in violations:
            if v["type"] == "forbidden_phrase":
                phrase = v["phrase"]
                replacement = self._get_replacement(phrase)
                revised = re.sub(re.escape(phrase), replacement, revised, flags=re.IGNORECASE)

            elif v["type"] == "missing_escalation" and risk_score >= 0.65:
                if "consult a doctor" not in revised.lower():
                    revised += "\\n\\nPlease consult a doctor or visit your nearest PHC for proper evaluation."

        # Always ensure disclaimer present
        disclaimer = "Yeh sirf wellness jankari hai. Kripya doctor se milein."
        if disclaimer not in revised:
            revised += f"\\n\\n{disclaimer}"

        return revised

    def _get_replacement(self, forbidden_phrase: str) -> str:
        replacements = {
            "you have": "I notice patterns that may suggest",
            "you are suffering from": "there are signs worth discussing regarding",
            "definitely": "possibly",
            "certainly indicates": "may suggest",
            "you need medication": "please consult your doctor about treatment options",
        }
        return replacements.get(forbidden_phrase.lower(),
                                "[please consult a doctor for this]")

    def _safe_fallback(self, risk_score: float) -> str:
        if risk_score >= 0.65:
            return ("Namaste Ji 🙏 I've noticed some health patterns that deserve attention. "
                    "Please consult a doctor or visit your nearest PHC as soon as you can. "
                    "Your family and Aarogya are here to support you. "
                    "\\n\\nYeh sirf wellness jankari hai. Kripya doctor se milein.")
        return ("Namaste Ji 🙏 Your wellness patterns look mostly stable. "
                "Keep monitoring and stay hydrated. If you feel unwell, please consult a doctor. "
                "\\n\\nYeh sirf wellness jankari hai. Kripya doctor se milein.")

CI/CD Pipeline

5-stage pipeline: unit tests → ML validation (AUC ≥ 0.88 + bias audit) → Docker build → staging E2E → production canary (10% traffic, 10min monitor, auto-rollback).

Pipeline Stages
1
Test
pytest unit + integration + constitutional AI tests. Coverage uploaded to Codecov. Runs on all pushes to main/develop.
2
ML Model Validation
validate_model.py — AUC must be ≥ 0.88. bias_eval.py — max AUC gap across demographics must be ≤ 0.08. Both must pass before build.
3
Build & Push Docker Images
Builds api and ml-worker images tagged with git SHA. Pushes to gcr.io/cardioguardai registry. Only runs on main branch.
4
Deploy to Staging
GKE cardioguard-staging in asia-south1. Rolling deployment with 300s timeout. E2E tests against https://staging-api.cardioguardai.in.
5
Production Canary Deploy
Routes 10% traffic to canary deployment. Monitors error rate for 10 minutes. If error_rate < 1%: promote to 100%. On failure: auto-rollback via kubectl rollout undo.
Source: cicd.yml
.github/workflows/cicd.ymlYAML · GitHub Actions
name: CardioGuardAI CI/CD Pipeline

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

env:
  GCR_REGISTRY: gcr.io/cardioguardai
  GKE_CLUSTER: cardioguard-prod
  GKE_REGION: asia-south1

jobs:
  # ── 1. Test ──────────────────────────────────────────────────
  test:
    name: Run Tests
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.11'

      - name: Install dependencies
        run: |
          pip install -r src/api/requirements.txt
          pip install -r src/ml/requirements.txt
          pip install pytest pytest-asyncio pytest-cov

      - name: Run unit tests
        run: pytest tests/unit/ -v --cov=src --cov-report=xml

      - name: Run integration tests
        run: pytest tests/integration/ -v
        env:
          TEST_DATABASE_URL: postgresql://test:test@localhost:5432/test

      - name: Constitutional AI tests
        run: pytest tests/unit/test_constitutional.py -v

      - name: Upload coverage
        uses: codecov/codecov-action@v3
        with:
          file: coverage.xml

  # ── 2. ML Model Validation ───────────────────────────────────
  ml-validation:
    name: ML Model Validation
    runs-on: ubuntu-latest
    needs: test
    steps:
      - uses: actions/checkout@v4

      - name: Validate model AUC
        run: |
          python src/ml/evaluation/validate_model.py \\
            --checkpoint config/model.pt \\
            --test-data data/processed/test.parquet \\
            --min-auc 0.88
        env:
          MLFLOW_TRACKING_URI: \${{ secrets.MLFLOW_URI }}

      - name: Bias audit
        run: |
          python src/ml/evaluation/bias_eval.py \\
            --max-auc-gap 0.08

  # ── 3. Build Docker Images ───────────────────────────────────
  build:
    name: Build & Push Images
    runs-on: ubuntu-latest
    needs: [test, ml-validation]
    if: github.ref == 'refs/heads/main'
    steps:
      - uses: actions/checkout@v4

      - name: Authenticate to GCR
        uses: google-github-actions/auth@v1
        with:
          credentials_json: \${{ secrets.GCP_SA_KEY }}

      - name: Build API image
        run: |
          docker build -t $GCR_REGISTRY/api:\${{ github.sha }} ./src/api
          docker push $GCR_REGISTRY/api:\${{ github.sha }}

      - name: Build ML Worker image
        run: |
          docker build -t $GCR_REGISTRY/ml-worker:\${{ github.sha }} ./src/ml
          docker push $GCR_REGISTRY/ml-worker:\${{ github.sha }}

  # ── 4. Deploy to Staging ─────────────────────────────────────
  deploy-staging:
    name: Deploy to Staging
    runs-on: ubuntu-latest
    needs: build
    environment: staging
    steps:
      - uses: actions/checkout@v4

      - name: Deploy to GKE staging
        run: |
          gcloud container clusters get-credentials cardioguard-staging --region $GKE_REGION
          kubectl set image deployment/cardioguard-api api=$GCR_REGISTRY/api:\${{ github.sha }}
          kubectl rollout status deployment/cardioguard-api --timeout=300s

      - name: Run E2E tests on staging
        run: pytest tests/e2e/ -v --base-url=https://staging-api.cardioguardai.in

  # ── 5. Deploy to Production (canary) ─────────────────────────
  deploy-production:
    name: Deploy to Production (Canary)
    runs-on: ubuntu-latest
    needs: deploy-staging
    environment: production
    steps:
      - uses: actions/checkout@v4

      - name: Deploy canary (10% traffic)
        run: |
          gcloud container clusters get-credentials $GKE_CLUSTER --region $GKE_REGION
          kubectl set image deployment/cardioguard-api-canary api=$GCR_REGISTRY/api:\${{ github.sha }}
          # Send 10% traffic to canary
          kubectl patch virtualservice cardioguard-vs --patch '{"spec":{"http":[{"match":[{"headers":{"canary":{"exact":"true"}}}],"route":[{"destination":{"host":"cardioguard-api-canary"}}]},{"route":[{"destination":{"host":"cardioguard-api"},"weight":90},{"destination":{"host":"cardioguard-api-canary"},"weight":10}]}]}}'

      - name: Monitor canary for 10min
        run: |
          sleep 600
          python scripts/check_canary_metrics.py --threshold-error-rate 0.01

      - name: Promote canary to 100%
        if: success()
        run: |
          kubectl set image deployment/cardioguard-api api=$GCR_REGISTRY/api:\${{ github.sha }}
          kubectl rollout status deployment/cardioguard-api

      - name: Rollback on failure
        if: failure()
        run: |
          kubectl rollout undo deployment/cardioguard-api-canary
          echo "Canary rollback triggered — previous version restored"
